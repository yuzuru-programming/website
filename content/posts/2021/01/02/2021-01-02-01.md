### 3 ソースコードを書く

---

{{< code lang="javascript" title="gas.js" >}}
function myFunction() {
  const html = UrlFetchApp.fetch('https://www.yahoo.co.jp').getContentText('UTF-8');

  // jqueryチックに使えるように変換
  const $ = Cheerio.load(html);

  const _li = $('main article section ul')
    .eq(0)
    .find('li');

  // ヤフートップニュースを表示
  _li.map(function(i) {
    console.log(_li.eq(i).text());
    console.log(
      _li
        .eq(i)
        .find('a')
        .attr()['href']
    );
    console.log();
  });
}
